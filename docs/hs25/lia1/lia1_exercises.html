import numpy as np

M = np.array([[1, 7], [4, 2]])
M
array([[1, 7],
       [4, 2]])
N = np.diag([6, 5])
N
array([[6, 0],
       [0, 5]])
P = np.full(shape=[2, 2], fill_value=-np.pi)
P
array([[-3.14159265, -3.14159265],
       [-3.14159265, -3.14159265]])
Q = np.e * np.eye(1) + np.sqrt(2) * np.eye(2)

array([[4.13249539, 2.71828183],

R = np.block([[M, N], [P, Q]])
